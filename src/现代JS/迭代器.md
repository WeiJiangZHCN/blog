# 迭代器

看个例子

```JS
function num() {
    return {
        next:()=> {
            return this.min < this.max 
            ? {value:this.min++,done:false} 
            : {value:undefined,done:true}
        }
    }
}
let obj = {
    min:1,
    max:10,
    [Symbol.iterator]:num
}
for (let i of obj) {
    console.log(i)
}
```
